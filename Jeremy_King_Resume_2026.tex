% CUSTOM COMMANDS --------------

% Command 'smallpipe'
% A slightly shorter | character for a nicer look as a section header segment separator.
\newcommand{\smallpipe}{\textbf{\raisebox{0.2ex}{\scalebox{1}[0.8]{\textbar}}}}

% Command 'resumeItem'
% A regular, left-aligned, indented bullet point for use in multiple sections.
\newcommand{\resumeItem}[1]{
	\item\small{{#1 \vspace{-2pt}}}
}

% Command 'resumeDoubleSubheading'
% A section containing one primary header and two secondary headers.
% The primary header has one bold left-aligned segment and one regular right-aligned segment, in that order.
% The secondary headers have one italicized left-aligned segment and one italicized right-aligned segment, in that order.
\newcommand{\resumeDoubleSubheading}[7]{
	\vspace{-2pt}
	\item
		\begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
		\textbf{#1} \smallpipe{} \textit{\small{GPA #2}} & #3 \\
		\textit{\small#4} & \textit{\small #5} \\
		\textit{\small#6} & \textit{\small #7} \\
		\end{tabular*}\vspace{-7pt}
}

% Command 'resumeSubheading'
% A section containing one primary header and one secondary header.
% The primary header has one bold left-aligned segment, one bold separator (smallpipe), one italicized left-aligned segment, and one regular right-aligned segment, in that order.
% The secondary header has one italicized left-aligned segment and one italicized right-aligned segment, in that order.
\newcommand{\resumeSubheading}[5]{
	\vspace{-2pt}
	\item
		\begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
		\textbf{#1} \smallpipe{} \textit{\small#2} & #3 \\
		\textit{\small#4} & \textit{\small #5} \\
		\end{tabular*}\vspace{-7pt}
}

% Command 'resumeSubSubheading'
% A secondary header with one italicized left-aligned segment and one right-aligned segment.
\newcommand{\resumeSubSubheading}[2]{
	\item
		\begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
		\textit{\small#1} & \textit{\small #2} \\
		\end{tabular*}\vspace{-7pt}
}

% Command 'resumeProjectHeading'
% A section containing one primary header.
% The primary header has one bold left-aligned segment, one bold separator (smallpipe), one italicized left-aligned segment, and one regular right-aligned segment, in that order.
\newcommand{\resumeProjectHeading}[3]{
	\item
		\begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
		\textbf{#1} \smallpipe{} \textit{\small#2} & #3 \\
		\end{tabular*}\vspace{-7pt}
}

% Command 'resumeProjectHeadingWebsite'
% A section containing one primary header.
% The primary header has one bold left-aligned segment, one bold separator (smallpipe), one italicized hyperlink segment, one bold separator, one italicized left-aligned segment, and one regular right-aligned segment, in that order.
\newcommand{\resumeProjectHeadingWebsite}[4]{
	\item
		\begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
		\textbf{#1} \smallpipe{} \small{\href{#2}{Link}} \smallpipe{} \textit{\small#3} & #4 \\
		\end{tabular*}\vspace{-7pt}
}
